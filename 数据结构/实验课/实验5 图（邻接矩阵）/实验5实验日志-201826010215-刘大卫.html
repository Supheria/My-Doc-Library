  • 需要注意的是，为了可以实现访问我们定义的是类指针：

```

#include "Graph.h"
#include "Graphl.h"
using namespace std;
int main(){
    Graphm<char> *mmp = new Graphm<char>;
    int m = 0, n = 0, wt = 0;
    char va, vb;
    cin >> n >> m;
    char a[n];
    for(int i = 0;i < n;i ++) {
        cin >> a[i];
        mmp->setPoint(a[i]);
    }
    for(int i = 0;i < m;i ++) {
        cin >> va >> vb >> wt;
        mmp->setEdge(mmp->val(va).second, mmp->val(vb).second, wt);
    }
    mmp->print();
    return 0;
}

```
